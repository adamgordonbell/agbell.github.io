Project Euler 8 in C#

November 10, 2010 - euler c#

Discover the largest product of five consecutive digits in the 1000-digit number.

http://projecteuler.net/index.php?section=problems&id=8

[TestClass]
    public class Problem8
    {

        [TestMethod]
        public void Answer()
        {
            var Consecutive5s =
                from a in 0.To(999 - 5)
                select new
                {
                    sum = ConsecutiveFive(a)
                        .Select(x => Convert.ToInt32(x))
                        .Product(),
                    numbers = ConsecutiveFive(a)
                };

            var max = Consecutive5s.First(x => x.sum == Consecutive5s.Max(y => y.sum));
            Console.WriteLine(max.sum);
        }

        static IEnumerable<Char> ConsecutiveFive(int start)
        {
            return numbers
                        .AsEnumerable()
                        .Skip(start)
                        .Take(5);

        }
